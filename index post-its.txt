    <script>
        const SCRIPT_URL = "https://script.google.com/macros/s/AKfycbyd8EMwdnW3MVat7v7zWbj5n0CHM5sl8Ket0HZcKdEVAqGysjMqH0POzgTA6NX2LHue3Q/exec";
        const apiKey = ""; 

        let dadesTotals = [];
        let escolesUniques = [];
        let escolesActives = new Set();
        let clauEscola = "";

        const SECCIONS = [
            { títol: "M'ha agradat...", icona: "❤️" },
            { títol: "He après...", icona: "🎓" },
            { títol: "M'ha interessat...", icona: "🌟" },
            { títol: "Dificultats", icona: "🧩" },
            { títol: "Dubtes", icona: "🤔" }
        ];

        async function carregarDades() {
            const loading = document.getElementById('loading');
            loading.classList.remove('hidden');
            try {
                const res = await fetch(SCRIPT_URL + "?t=" + Date.now());
                dadesTotals = await res.json();
                if (dadesTotals.length > 0) {
                    const claus = Object.keys(dadesTotals[0]);
                    clauEscola = claus.find(k => k.toLowerCase().includes('centre') || k.toLowerCase().includes('escola')) || claus[1];
                    escolesUniques = [...new Set(dadesTotals.map(d => d[clauEscola]).filter(Boolean))].sort();
                    if (escolesActives.size === 0) { escolesUniques.forEach(e => escolesActives.add(e)); }
                    actualitzarApp();
                }
            } catch (e) { console.error("Error:", e); }
            finally { loading.classList.add('hidden'); }
        }

        function actualitzarApp() {
            document.getElementById('statEscoles').innerText = escolesUniques.length;
            document.getElementById('statRespostes').innerText = dadesTotals.length;
            renderFiltres();
            renderMural();
        }

        function renderFiltres() {
            const container = document.getElementById('filterContainer');
            container.innerHTML = '';
            escolesUniques.forEach(escola => {
                const actiu = escolesActives.has(escola);
                const btn = document.createElement('button');
                btn.className = `btn-escola px-6 py-2.5 rounded-2xl text-sm font-bold shadow-sm ${actiu ? 'active' : 'bg-white text-slate-400 hover:bg-slate-50 border-slate-100'}`;
                btn.innerText = escola;
                btn.onclick = () => {
                    if (escolesActives.has(escola)) { if (escolesActives.size > 1) escolesActives.delete(escola); }
                    else { escolesActives.add(escola); }
                    actualitzarApp();
                };
                container.appendChild(btn);
            });
        }

        function renderMural() {
            const grid = document.getElementById('muralGrid');
            const empty = document.getElementById('emptyState');
            grid.innerHTML = '';
            if (escolesActives.size === 0) { empty.classList.remove('hidden'); return; }
            empty.classList.add('hidden');
            const claus = Object.keys(dadesTotals[0]);

            Array.from(escolesActives).forEach(escolaNom => {
                const respostesEscola = dadesTotals.filter(d => d[clauEscola] === escolaNom);
                const section = document.createElement('section');
                section.className = "mural-section p-8 md:p-14 mb-16 shadow-sm";
                section.innerHTML = `
                    <div class="flex flex-col md:flex-row justify-between items-center mb-12 gap-6">
                        <div class="text-center md:text-left">
                            <h2 class="title-font text-4xl text-slate-800 italic underline decoration-yellow-300 decoration-8 underline-offset-8">${escolaNom}</h2>
                            <p class="text-slate-400 text-sm mt-4 font-bold uppercase tracking-widest">${respostesEscola.length} idees recollides</p>
                        </div>
                        <button onclick="ferResumIA('${escolaNom.replace(/'/g, "\\'")}')" class="bg-yellow-400 hover:bg-yellow-500 text-slate-900 px-8 py-3 rounded-2xl font-black shadow-lg transition-all flex items-center gap-2 text-sm uppercase tracking-tight">
                            <span>✨ Foto de les veus</span>
                        </button>
                    </div>
                    <div class="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-5 gap-8">
                        ${SECCIONS.map((sec, i) => {
                            const textos = respostesEscola.map(r => r[claus[i + 2]]).filter(t => t && t.toString().trim().length > 1);
                            return `
                                <div class="flex flex-col">
                                    <div class="text-center mb-6">
                                        <div class="text-4xl mb-2">${sec.icona}</div>
                                        <h4 class="title-font text-[11px] uppercase font-black text-slate-400 tracking-widest">${sec.títol}</h4>
                                    </div>
                                    <div class="flex flex-col gap-4">
                                        ${textos.length > 0 ? textos.map(txt => `<div class="post-it col-${i}"><p class="handwritten text-xl text-slate-700 leading-tight">${txt}</p></div>`).join('') : '<p class="text-center text-slate-200 italic text-xs py-8">Sense dades</p>'}
                                    </div>
                                </div>
                            `;
                        }).join('')}
                    </div>
                `;
                grid.appendChild(section);
            });
        }

        async function ferResumIA(nomEscola) {
            const modal = document.getElementById('reportModal');
            const content = document.getElementById('reportContent');
            const btnPdf = document.getElementById('btnDownloadPDF');
            modal.classList.remove('hidden');
            btnPdf.classList.add('hidden');
            content.innerHTML = `<div class="text-center py-20"><div class="loader inline-block rounded-full border-4 h-12 w-12 mb-6"></div><p class="handwritten text-3xl text-slate-500">Revelant veus de ${nomEscola}...</p></div>`;

            const respostes = dadesTotals.filter(d => d[clauEscola] === nomEscola);
            const promptDades = respostes.map(r => {
                const c = Object.keys(r);
                return `[S: ${r[c[2]]}, ${r[c[3]]}, ${r[c[4]]}, ${r[c[5]]}, ${r[c[6]]}]`;
            }).join('\n');

            try {
                const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        contents: [{ parts: [{ text: `
                        Ets un sintetitzador d'idees per a l'escola ${nomEscola}.
                        
                        OBJECTIU: Crea una fotografia escrita de les veus dels nens. 
                        
                        INSTRUCCIONS DE FORMAT:
                        - No posis introduccions ni comiats.
                        - Els títols d'apartat han d'anar en NEGRETA (usa <b>Títol</b>).
                        - Hi ha d'haver un SALT DE LÍNIA clar (amb <br>) entre el títol i el resum/descripció.
                        - Agrupa els temes recurrents si n'hi ha molts.
                        - Usa <div class="ia-quote">"frase"</div> per a 1 o 2 cites textuals originals per secció.
                        
                        APARTATS:
                        1. <b>❤️ M'ha agradat...</b><br>
                        2. <b>🎓 He après...</b><br>
                        3. <b>🌟 M'ha interessat...</b><br>
                        4. <b>🧩 Dificultats</b><br>
                        5. <b>🤔 Dubtes</b><br>
                        
                        DADES:
                        ${promptDades}` }] }]
                    })
                });

                const json = await response.json();
                let textIA = json.candidates?.[0]?.content?.parts?.[0]?.text || "No s'ha pogut generar el resum.";
                
                content.innerHTML = `
                    <div id="pdfArea">
                        <div class="mb-10 text-center">
                            <h2 class="title-font text-4xl text-slate-800 mb-2">${nomEscola}</h2>
                            <p class="handwritten text-xl text-slate-500">Fotografia del procés d'aprenentatge • ${new Date().toLocaleDateString('ca-ES')}</p>
                            <div class="h-1 w-20 bg-yellow-400 mx-auto mt-4 rounded-full"></div>
                        </div>
                        <div class="space-y-8">
                            ${textIA.replace(/#{1,6}\s?/g, '').replace(/\*/g, '')}
                        </div>
                    </div>
                `;
                btnPdf.classList.remove('hidden');
            } catch (e) { content.innerHTML = `<p class="text-center py-20 text-red-400">Error en el sistema. Torna-ho a provar.</p>`; }
        }

        function tancarModal() { document.getElementById('reportModal').classList.add('hidden'); }
        function descarregarPDF() {
            const element = document.getElementById('pdfArea');
            const escolaTitol = element.querySelector('h2').innerText;
            const opt = { margin: 15, filename: `Veus_${escolaTitol}.pdf`, image: { type: 'jpeg', quality: 1 }, html2canvas: { scale: 3, useCORS: true }, jsPDF: { unit: 'mm', format: 'a4', orientation: 'portrait' } };
            html2pdf().from(element).set(opt).save();
        }
        window.onload = carregarDades;
    </script>
